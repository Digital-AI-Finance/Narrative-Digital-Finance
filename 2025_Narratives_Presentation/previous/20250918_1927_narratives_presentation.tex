\begin{frame}{Mathematical Framework: Intensity Measures}
\begin{block}{Narrative Intensity}
$$I_{n,t} = \frac{\text{Articles relevant to narrative } n \text{ at time } t}{\text{Total articles at time } t}$$
\end{block}

\begin{block}{Negative Intensity}
$$NI_{n,t} = \frac{\text{Negative sentiment articles for narrative } n}{\text{Total articles at time } t}$$
\end{block}

\begin{block}{7-Day Rolling Average}
$$\bar{NI}_{n,t} = \frac{1}{7} \sum_{i=0}^{6} NI_{n,t-i}$$
\end{block}

\textbf{Weekly Change:} $\Delta \bar{NI}_{n,t} = \bar{NI}_{n,t} - \bar{NI}_{n,t-7}$
\end{frame}

\begin{frame}{Regression Methodology}
\begin{block}{Contemporaneous Analysis}
$$R_{SPY,t} = \alpha + \beta \cdot \Delta \bar{NI}_{n,t} + \epsilon_t$$
\end{block}

\begin{block}{Predictive Regression}
$$R_{t+1} = \alpha + \beta_1 \Delta NI_{n,t} + \beta_2 VIX_t + \beta_3 R_t + \beta_4 R_{t-1} + \epsilon_t$$
\end{block}

\textbf{Rolling Window:} 3-month univariate regressions\\
\textbf{Standardization:} 60-day rolling z-scores\\
\textbf{HAC Standard Errors:} Newey-West adjustment
\end{frame}

\begin{frame}{Advanced Econometric Models}
\begin{block}{Vector Autoregression (VAR)}
$$\begin{bmatrix} R_t \\ NI_{1,t} \\ \vdots \\ NI_{k,t} \end{bmatrix} = c + \sum_{i=1}^{p} \Phi_i \begin{bmatrix} R_{t-i} \\ NI_{1,t-i} \\ \vdots \\ NI_{k,t-i} \end{bmatrix} + \epsilon_t$$
\end{block}

\begin{block}{Narrative Factor Model}
$$R_{i,t} = \alpha_i + \sum_{j=1}^{K} \beta_{i,j} F_{j,t}^{narrative} + \sum_{m=1}^{M} \gamma_{i,m} F_{m,t}^{traditional} + \epsilon_{i,t}$$
\end{block}

where $F^{narrative}$ are narrative factors extracted via PCA
\end{frame}

\begin{frame}{Portfolio Optimization with Narrative Constraints}
\begin{block}{Mean-Variance with Narrative Exposure}
$$\max_w \left\{ w^T \mu - \frac{\lambda}{2} w^T \Sigma w + \gamma \sum_{j} \alpha_j E_j(w) \right\}$$
\end{block}

Subject to:
\begin{itemize}
\item $\sum_i w_i = 1$ (fully invested)
\item $|E_j(w)| \leq \bar{E}_j$ (narrative exposure limits)
\item $w_i \geq 0$ (long-only constraint)
\end{itemize}

where $E_j(w) = \sum_i w_i \beta_{i,j}^{narrative}$ is portfolio narrative exposure

\vspace{0.3cm}
\textbf{Implementation:} Quadratic programming with narrative beta constraints
\end{frame}
