% Methodology Overview
\begin{frame}{Data Pipeline: From Media to Market Signals}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Data Collection}
\begin{itemize}
\item 150,000+ global digital media sources
\item Articles tagged for 73 narratives
\item NLP-based sentiment scoring
\item Daily intensity measurements
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Key Metrics}
\begin{itemize}
\item \textbf{Intensity}: Proportion of articles per narrative
\item \textbf{Negative Intensity}: Articles with negative sentiment
\item \textbf{7-day rolling averages} for stability
\item Weekly changes for market impact
\end{itemize}
\end{column}
\end{columns}
\vspace{\fill}
{\footnotesize \color{annotation} Processing pipeline runs continuously from July 2015 to present}

% Market Crash Narrative Analysis
\begin{frame}{Market Crash Narrative: The Dominant Driver}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Regression Results}
\begin{itemize}
\item Coefficient: \textbf{-0.26}
\item T-statistic: \textbf{-9.94}

\begin{itemize}

\begin{itemize}

\textbf{Practical Application}
\begin{itemize}

\item 2-week holding period
\item Outperforms 50/50 benchmark
\end{itemize}

% COVID Recovery Portfolio
\begin{frame}{COVID-19 Recovery Portfolio Construction}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Methodology}
\begin{itemize}
\item Estimate stock-level narrative betas
\item Select 25 lowest beta stocks (long)
\item Select 25 highest beta stocks (short)
\item Monthly rebalancing
\end{itemize}
\vspace{0.5cm}
\textbf{Low Beta Examples}
\begin{itemize}
\item Wynn Resorts (-2.98)
\item Disney (-2.92)
\item Las Vegas Sands (-2.43)
\item Halliburton (-3.10)
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Performance Results}
\begin{itemize}
\item Pre-vaccine: \textbf{-32.25\%}
\item Post-vaccine: \textbf{+120.74\%}
\item Total period: +88.49\%
\item Information Ratio: 2.01
\end{itemize}
\vspace{0.5cm}
\textbf{vs. Case-Count Strategy}
\begin{itemize}
\item Case-based: +16.55\% only
\item Narrative captures sentiment better
\item Media influence > raw statistics
\end{itemize}
\end{column}
\end{columns}
\vspace{\fill}
{\footnotesize \color{annotation} Portfolio pivoted November 9, 2020 (Pfizer vaccine announcement)}
\end{frame}
